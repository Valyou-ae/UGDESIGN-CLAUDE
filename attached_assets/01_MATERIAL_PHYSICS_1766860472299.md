### Basic Material Enhancement

```typescript
function enhanceWithMaterial(prompt: string, material: string): string {
  const materialData = getMaterialProfile(material);
  
  return `${prompt}, ${materialData.promptKeywords.join(', ')}, 
    physically accurate ${material} rendering`;
}

// Example:
// Input: "luxury watch on table"
// Output: "luxury watch on table, polished gold with mirror-like 
//          reflections, warm yellow metallic sheen, bright specular 
//          highlights, physically accurate polished gold rendering"
```

### Multi-Material Scene

```typescript
function enhanceMultiMaterial(prompt: string, materials: string[]): string {
  const enhancements = materials.map(m => {
    const data = getMaterialProfile(m);
    return data.promptKeywords.slice(0, 3).join(', ');
  });
  
  return `${prompt}, ${enhancements.join(', ')}, 
    realistic material interactions, 
    physically accurate surfaces`;
}

// Example with gold watch, leather strap, glass crystal:
// "luxury watch, polished gold with mirror-like reflections, 
//  genuine leather with natural grain texture, crystal clear 
//  glass with sharp refractions, realistic material interactions"
```
