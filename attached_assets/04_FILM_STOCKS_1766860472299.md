### Film Stock Application

```typescript
function applyFilmStock(prompt: string, stock: string): string {
  const film = getFilmStock(stock);
  
  return `${prompt}, ${film.promptKeywords.join(', ')}, 
    authentic ${stock} film emulation`;
}

// Example:
// Input: "portrait of musician in studio"
// Stock: "Portra 400"
// Output: "portrait of musician in studio, Kodak Portra 400 
//          color profile, muted pastel film tones, flattering 
//          natural skin tones, fine film grain texture"
```
