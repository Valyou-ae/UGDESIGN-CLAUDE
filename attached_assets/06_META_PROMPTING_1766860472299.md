```typescript
function buildOptimizedPrompt(elements: PromptElements): string {
  const positive = [
    elements.subject,
    elements.action,
    elements.setting,
    elements.lighting,
    elements.style,
    elements.technical,
    qualityKeywords.resolution.slice(0, 2).join(', '),
    qualityKeywords.photography[0]
  ].filter(Boolean).join(', ');
  
  const negative = [
    ...universalNegatives,
    ...(elements.subject.includes('portrait') ? portraitNegatives : [])
  ].join(', ');
  
  return { positive, negative };
}
```
