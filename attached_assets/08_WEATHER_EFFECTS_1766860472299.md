```typescript
function applyWeatherAtmosphere(prompt: string, weather: Weather): string {
  const effect = weatherEffects[weather.type];
  const time = timeOfDay[weather.timeOfDay];
  const season = seasons[weather.season];
  
  return `${prompt}, ${effect.promptKeywords.join(', ')}, 
    ${time.promptKeywords.slice(0, 2).join(', ')}, 
    ${season.promptKeywords[0]}`;
}
```
