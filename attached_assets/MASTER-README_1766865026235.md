## ğŸ”§ INTEGRATION EXAMPLES

### Portrait Photography Service

```typescript
import HumanAnatomy from './tier-1/01-human-anatomy-perfection';
import LensSimulation from './tier-1/02-photographic-lens-simulation';
import Imperfections from './tier-1/03-imperfection-library';
import Lighting from './tier-1/04-advanced-lighting-physics';
import Materials from './tier-1/05-surface-material-accuracy';

class PortraitService {
  generateProfessionalPortrait(options: {
    gender: 'male' | 'female';
    lighting: string;
    mood: string;
  }): { positive: string; negative: string } {
    
    // Get anatomy profile
    const anatomy = HumanAnatomy.getAnatomyProfile(options.gender);
    
    // Get recommended lens
    const lens = LensSimulation.recommendLens('portrait');
    
    // Get lighting for mood
    const lighting = Lighting.getLightingForMood(options.mood);
    
    // Get imperfections
    const imperfections = Imperfections.generateBalancedImperfections('natural_portrait');
    
    // Get skin material
    const skin = Materials.generateEnhancedSkinPrompt('fair');
    
    // Build prompts
    const positive = [
      `Professional ${options.gender} portrait`,
      HumanAnatomy.generateAnatomyPrompt(anatomy!),
      LensSimulation.generateLensPrompt(lens!, 'f/1.4'),
      Lighting.generateLightingPrompt(lighting!),
      imperfections,
      skin
    ].join(', ');
    
    const negative = HumanAnatomy.generateAnatomyNegativePrompt();
    
    return { positive, negative };
  }
}
```

### Landscape Photography Service

```typescript
import DepthSystems from './enhancement-features/07-advanced-depth-systems';
import Weather from './enhancement-features/08-weather-atmospheric-effects';
import SubjectLibraries from './enhancement-features/05-subject-specific-libraries';
import ColorPsychology from './enhancement-features/03-color-psychology-library';

class LandscapeService {
  generateEpicLandscape(options: {
    weather: string;
    mood: string;
    timeOfDay: string;
  }): string {
    
    const depth = DepthSystems.getDepthProfile('deep_landscape');
    const weather = Weather.getWeatherCondition(options.weather);
    const palette = ColorPsychology.getPaletteForMood(options.mood);
    const landscape = SubjectLibraries.getSubjectLibrary('landscape');
    
    return [
      'Epic landscape photography',
      DepthSystems.generateDepthPrompt(depth!),
      Weather.generateWeatherPrompt(weather!),
      ColorPsychology.generateEmotionalPalettePrompt(palette!),
      landscape?.promptTemplates[0]
    ].filter(Boolean).join(', ');
  }
}
```
