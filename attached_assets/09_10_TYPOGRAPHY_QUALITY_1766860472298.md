const qualityScore = {
  
  technical: {
    weight: 30,
    criteria: ["Resolution", "Sharpness", "Noise", "Artifacts"]
  },
  
  anatomical: {
    weight: 25,
    criteria: ["Hands", "Face", "Body", "Proportions"]

    weight: 25,
    criteria: ["Lighting", "Shadows", "Color", "Style"]
  },
  
  composition: {
    weight: 20,
    criteria: ["Framing", "Balance", "Flow", "Subject"]
  },
  
  calculateScore: (scores: Record<string, number>) => {
    return Object.entries(scores).reduce((total, [key, value]) => {
      return total + (value * qualityScore[key].weight / 100);
    }, 0);
  }
};
