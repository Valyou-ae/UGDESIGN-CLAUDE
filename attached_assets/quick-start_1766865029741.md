## Step 3: Generate Your First Prompt

### Portrait Example

```typescript
// 1. Get the profiles you need
const anatomy = Anatomy.getAnatomyProfile('female', '20-35');
const lens = Lens.getLensProfile('canon_85mm_f12');
const lighting = Lighting.getLightingScenario('golden_hour');

// 2. Generate prompt parts
const anatomyPrompt = Anatomy.generateAnatomyPrompt(anatomy!);
const lensPrompt = Lens.generateLensPrompt(lens!, 'f/1.4');
const lightingPrompt = Lighting.generateLightingPrompt(lighting!);
const imperfectionPrompt = Imperfections.generateBalancedImperfections('natural_portrait');

// 3. Combine into final prompt
const prompt = `
Professional portrait of elegant woman,
${anatomyPrompt},
${lensPrompt},
${lightingPrompt},
${imperfectionPrompt}
`;

// 4. Get negative prompt
const negativePrompt = Anatomy.generateAnatomyNegativePrompt();

// 5. Use with your API
console.log('Positive:', prompt);
console.log('Negative:', negativePrompt);
```

## Step 4: Common Patterns

### Quick Portrait

```typescript
function quickPortrait(gender: 'male' | 'female') {
  const { positive, negative } = Anatomy.getCompletePortraitPrompt(gender);
  return { prompt: positive, negative_prompt: negative };
}
```

### Quick Landscape

```typescript
import Depth from './enhancement-features/07-advanced-depth-systems';
import Weather from './enhancement-features/08-weather-atmospheric-effects';

function quickLandscape(weather: string = 'golden_hour') {
  const depth = Depth.generateDepthPrompt(Depth.getDepthProfile('deep_landscape')!);
  const weatherPrompt = Weather.generateWeatherPrompt(Weather.getWeatherCondition(weather)!);
  
  return `Epic landscape, ${depth}, ${weatherPrompt}`;
}
```

### Quick Product Shot

```typescript
import Materials from './enhancement-features/01-material-physics-library';

function quickProduct(material: string) {
  const mat = Materials.getMaterial(material);
  return mat ? Materials.generateMaterialPrompt(mat) : '';
}
```
